数据结构实习0 海龟作图

发表于 2017-11-23   |   分类于 数据结构

0.3海龟作图

【问题描述】设计并实现海龟抽象数据类型Turtle，并以此为基础设计一个演示海龟作图的程序。
要解决这个问题需要你对图形库有一定的了解。如果在这之前你还未接触过图形库你可以点击这里
来了解以下easyX图形库。
首先根据题目要求可以写出Turtle的数据结构如下

typedef struct aPoint
{
    float oldx;'
    float oldy;
    float newx;
    float newy;
};
typedef struct penState
{
    bool up;
};
class Turtle
{
    public:
    void StartTurtle();
    void PenUp();
    void PenDown();
    bool isPenUp();
    float TurtleHeading();
    aPoint TurtlePos();
    void Move(int steps);
    void Turn(float degress);
    void MoveTTo(aPoint newPos);
    void TurnTTo(float angle);
    void SetTurtleColor(int color);
    aPoint pos;
    float heading;
    penState penstate;
};

这些函数和变量基本是按照书中所说的进行设计和编写的。
接下来就是函数的实现了

void Turtle::StartTurtle()
{
    pos.oldx = 0;
    pos.oldy = 0;
    pos.newx = WIDTH / 2;
    pos.newy = HEIGHT / 2;
}
void Turtle::PenUp()
{
    penstate.up = true;
}
void Turtle::PenDown()
{
    penstate.up = false;
}
bool Turtle::isPenUp()
{
    return penstate.up;
}
float Turtle::TurtleHeading()
{
    return heading;
}
aPoint Turtle::TurtlePos()
{
    return pos;
}
void Turtle::Move(int steps)
{
    pos.oldx = pos.newx;
    pos.oldy = pos.newy;
    pos.newx += steps*sin(PI/float(180)*TurtleHeading());
    pos.newy += steps*cos(PI/float(180)*TurtleHeading());
}
void Turtle::Turn(float degrees)
{
    heading += degrees;
    if (heading > 360)
        heading -= 360;
}
void Turtle::MoveTTo(aPoint newPos)
{
    pos.oldx = pos.newx;
    pos.oldy = pos.newy;
    pos.newx = newPos.newx;
    pos.newy = newPos.newy;
}
void Turtle::TurnTTo(float angle)
{
    heading = angle;
}
void Turtle::SetTurtleColor(int color)
{
    setlinecolor(color);
}

做完上述这些准备之后就可以根据题目要求进行实验了。

int main()
{
    initgraph(WIDTH, HEIGHT);
    Turtle turtle;
    turtle.heading = 90;
    turtle.PenDown();
    //turtle.SetTurtleColor(255);
    if (!turtle.isPenUp())
    {
        //画正方形
        aPoint newpos1;
        //第一根线
        newpos1.newx = 10;
        newpos1.newy = 10;
        turtle.MoveTTo(newpos1);
        newpos1.newx = 10;
        newpos1.newy = 100;
        turtle.MoveTTo(newpos1);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
        //第二根线
        newpos1.newx = 110;
        newpos1.newy = 100;
        turtle.MoveTTo(newpos1);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
        //第三根线
        newpos1.newx = 110;
        newpos1.newy = 10;
        turtle.MoveTTo(newpos1);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
        //第四根线
        turtle.Turn(180);
        turtle.Move(100);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
    }
    if (!turtle.isPenUp())
    {
        //画三角形
        aPoint newpos1;
        newpos1.newx = 250;
        newpos1.newy = 10;
        turtle.MoveTTo(newpos1);
        //
        turtle.TurnTTo(30);
        turtle.Move(100);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
        //
        turtle.TurnTTo(270);
        turtle.Move(100);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
        //
        turtle.TurnTTo(150);
        turtle.Move(100);
        line(turtle.pos.oldx, turtle.pos.oldy, turtle.pos.newx, turtle.pos.newy);
    }
    //不画了
    getchar();
    closegraph();
    return 0;

}
