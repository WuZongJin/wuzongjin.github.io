1.6池塘夜降彩色雨

【问题描述】设计一个程序，演示美丽的“池塘夜雨”的景色：色彩缤纷的雨点飘飘洒洒从天而降，滴入水有声，溅起
圈圈微波。

首先定义以下雨水的数据结构。

typedef struct Rain
{
    int x, y;//雨滴的位置
    int length;//雨滴的长度
    int path;//雨滴路径的长度
    bool drow;//雨滴是否落入池塘
    int speed;//雨滴的速度
    int width, height;//落入池塘溅起的水圈的宽和高
    int R, G, B;//颜色RGB
};
class ANIME
{
private:
    Rain rain[rNum];//rNum表示有多少雨滴
public:
    ANIME();
    void disturb(Rain &a, int size);//水圈逐渐变大
    void FrameFun(Rain &a);//落入池塘后更新水圈大小
    void reset(Rain &a);//重置雨滴
    void update();//更新动画
};

接下来是这些函数的实现

ANIME::ANIME()
{
    for (int i = 0; i < rNum; i++)
    {
        rain[i].length = rand() % 20 + 20;
        rain[i].path = rand() % 200 + 400;
        rain[i].x = rand() % WIDTH + 1;
        rain[i].y = 0;
        rain[i].drow = false;
        rain[i].speed = rand() % 20 + 10;
        rain[i].height = 5;
        rain[i].width = 10;
        rain[i].R = rand() % 100 + 50;
        rain[i].G = rand() % 100 + 50;
        rain[i].B = rand() % 100 + 50;
    }
}
void ANIME::disturb(Rain &a, int size)
{
    setlinecolor(RGB(a.R, a.G, a.B));
    ellipse(a.x-a.width,a.y-a.height, a.x + a.width, a.y + a.height);
    ellipse(a.x - a.width/2, a.y - a.height/2, a.x + a.width/2, a.y + a.height/2);
    a.width += size;
    a.height += size/2;
}
void ANIME::FrameFun(Rain &a)
{                
    disturb(a, 3);
    if (a.width >  100)
        reset(a);
}
void ANIME::update()
{
    for (int i = 0; i < rNum; i++)
    {
        if(!rain[i].drow)
        {
            setlinecolor(RGB(rain[i].R, rain[i].G, rain[i].B));
            line(rain[i].x, rain[i].y, rain[i].x, rain[i].y - rain[i].length);
            rain[i].y += rain[i].speed;
            if (rain[i].y > rain[i].path)
            {
                rain[i].drow = true;
            }
        }
        else FrameFun(rain[i]);
    }    
}
void ANIME::reset(Rain &a)
{
    a.length = rand() % 20 + 20;
    a.path = rand() % 200 + 400;
    a.x = rand() % WIDTH + 1;
    a.y = 0;
    a.drow = false;
    a.speed = rand() % 20 + 10;
    a.height = 5;
    a.width = 10;
    a.R = rand() % 50 + 50;
    a.G = rand() % 50 + 50;
    a.B = rand() % 50 + 50;
}

在这之后我们就可以在main函数中调用这个来进行动画演示

#include<iostream>
#include<graphics.h>
const int HEIGHT = 600;
const int WIDTH = 800;
const int rNum = 30;
void main()
{

    initgraph(WIDTH, HEIGHT);
    SetWindowText(GetHWnd(), "雨水");    
    setbkmode(TRANSPARENT);
    setlinecolor(WHITE);
    setfillcolor(BLACK);
    ANIME A;
    BeginBatchDraw();        
    while (true)
    {
        A.update();
        FlushBatchDraw();
        Sleep(60);
        fillrectangle(0, 0, WIDTH, HEIGHT);
    }
    EndBatchDraw();
}
